int rt[MAXM];

int build(int l, int r) {
    int p = ++tot;
    if (l == r) {
        t[p].val = l;
        t[p].rnk = 1;
        return p;

    int mid = (l + r) / 2;

int getRnk(int p, int l, int r, int pos) {  // æŸ¥è¯¢ç§©

    if (pos <= mid) {

        return getRnk(t[p].rc, mid + 1, r, pos);
