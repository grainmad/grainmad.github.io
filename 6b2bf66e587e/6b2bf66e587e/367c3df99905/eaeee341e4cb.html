字符串最小表示

string fun(string s) {   
    int a=0, b=1, c=0, n=s.size();
    while (a<n && b<n && c<n) {
        if (s[(a+c)%n] == s[(b+c)%n]) {
            c++;
        } else {
            if (s[(a+c)%n] > s[(b+c)%n]) a += c+1;
            else b += c+1;
            if (a == b) a++;
            c=0;
        }
    }
    a = min(a, b);
    return s.substr(a)+s.substr(0, a);
}
