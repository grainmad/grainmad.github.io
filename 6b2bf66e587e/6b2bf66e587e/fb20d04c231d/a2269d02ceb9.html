function<int(int,int)> dfs = [&](int x, int fa) { // 当前节点x, x的父节点fa
    int rt = nums[x];//当前节点值nums[x]
    for (int i:g[x]) {
        if (i == fa) continue;
        rt += dfs(i, x);
    }
    return rt;
};
